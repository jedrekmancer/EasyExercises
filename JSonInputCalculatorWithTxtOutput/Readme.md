calculates values depending on input and returns
results into file output.txt

{
"obj1": {
"operator": "add",
"value1": 2,
"value2": 3
},
"obj2": {
"operator": "sqrt",
"value1": 16
},
"obj3": {
"operator": "sqrt",
"value1": 36,
},
"obj4": {
"operator": "sub",
"value1": 43,
"value2": 12
},
"obj5": {
"operator": "add",
"value1": 2,
"value2": 3
},
